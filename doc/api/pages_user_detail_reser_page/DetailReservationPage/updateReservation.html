void updateReservation(int index) {
  Reservation currentReservation = reservationList[index];

  stdout.write(
      'Enter Name (or press Enter to keep current ${currentReservation.name}): ');
  String newName = stdin.readLineSync() ?? currentReservation.name;

  stdout.write(
      'Enter Phone Number (or press Enter to keep current ${currentReservation.phoneNumber}): ');
  String newPhoneNumber =
      stdin.readLineSync() ?? currentReservation.phoneNumber;

  stdout.write(
      'Enter Date (or press Enter to keep current ${currentReservation.date}): ');
  String newDate = stdin.readLineSync() ?? currentReservation.date;

  stdout.write(
      'Enter Time (or press Enter to keep current ${currentReservation.time}): ');
  String newTime = stdin.readLineSync() ?? currentReservation.time;

  stdout.write(
      'Enter Table (or press Enter to keep current ${currentReservation.table}): ');
  String newTable = stdin.readLineSync() ?? currentReservation.table;

  stdout.write(
      'Enter Number of Guests (or press Enter to keep current ${currentReservation.numberOfGuests}): ');
  int newNumberOfGuests = int.tryParse(stdin.readLineSync() ??
          currentReservation.numberOfGuests.toString()) ??
      currentReservation.numberOfGuests;

  List<ChoosenProduct> updatedChoosenProducts = [];

  for (ChoosenProduct chosenProduct in currentReservation.choosenProducts) {
    Product product = chosenProduct.product;

    stdout.write(
        'Enter the new count for ${product.nameOfProduct} (or press Enter to keep current count: ${chosenProduct.count}): ');
    int newCount = int.tryParse(
            stdin.readLineSync() ?? chosenProduct.count.toString()) ??
        chosenProduct.count;

    if (newCount > 0) {
      updatedChoosenProducts
          .add(ChoosenProduct(product: product, count: newCount));
    }
  }

  double newTotalPrice = calculateTotalPrice(updatedChoosenProducts);

  Reservation updatedReservation = Reservation(
    name: newName,
    phoneNumber: newPhoneNumber,
    date: newDate,
    time: newTime,
    choosenProducts: updatedChoosenProducts,
    numberOfGuests: newNumberOfGuests,
    table: newTable,
    totalPrice: newTotalPrice,
  );

  reservationList[index] = updatedReservation;
  print('Reservation updated successfully.');

  // Display the updated reservation details
  viewReservation(updatedReservation);
  Navigator.push(UserHomePage());
}
