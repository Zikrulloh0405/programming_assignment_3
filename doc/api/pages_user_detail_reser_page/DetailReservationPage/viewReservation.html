void viewReservation(Reservation reservation) {
  print('Reservation list ... :');

  for (int i = 0; i < reservationList.length; i++) {
    Reservation item = reservationList[i];
    print('${i + 1}. ${item.phoneNumber} - \$${item.totalPrice}');
  }

  print('Detail of Reservation $currentIndex');
  print('Name: ${reservation.name}');
  print('Phone Number: ${reservation.phoneNumber}');
  print('Date: ${reservation.date}');
  print('Time: ${reservation.time}');
  print('Table: ${reservation.table}');
  print('Number of Guests: ${reservation.numberOfGuests}');
  print('Total Price: \$${reservation.totalPrice.toStringAsFixed(2)}');

  print('Chosen Products:');
  for (int i = 0; i < reservation.choosenProducts.length; i++) {
    ChoosenProduct chosenProduct = reservation.choosenProducts[i];
    Product product = chosenProduct.product;
    print('Category: ${product.categoryOfProduct}');
    print('Name: ${product.nameOfProduct}');
    print('Price: \$${product.priceOfProduct.toStringAsFixed(2)}');
    print('Count: ${chosenProduct.count}');
    print(
        'Subtotal: \$${(product.priceOfProduct * chosenProduct.count).toStringAsFixed(2)}');
  }
}
