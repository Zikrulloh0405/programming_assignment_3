Future printEachResetvation(List<Reservation> listOfResevations) async {
  print("Reservation list ... : ");

  if (listOfResevations.isEmpty) {
    print('There is no reservations ... ');
    print('1. Create Reser.\n2. Return\n0. Exit');

    String userInput = stdin.readLineSync() ?? '0';
    switch (userInput) {
      case '1':
        Navigator.push(CreateReservation());
        break;
      case '2':
        Navigator.pop();
        break;
      case '0':
        exit(0);
      default:
    }
  } else {
    listOfResevations.asMap().forEach((index, item) {
      print('aaa');
      print("${index + 1}. ${item.phoneNumber} ${item.totalPrice}");
    });
  }

  print("Show details of reser ... (choose the number of reservation)");
  print("01. Create\n02. Return\n0. Exit");
  String userInput = stdin.readLineSync() ?? '0';
  int selectedIndex = int.tryParse(userInput) ?? 0;
  switch (userInput) {
    case '01':
      Navigator.push(CreateReservation());
      break;
    case '02':
      Navigator.pushReplacement(AdminHomePage());
      break;
    case '0':
      exit(0);

    default:
      if (selectedIndex > 0 && selectedIndex <= listOfResevations.length) {
        Reservation selectedReservation =
            listOfResevations[selectedIndex - 1];
        await Navigator.push(
            DetailReservationPage(selectedReservation, selectedIndex));
      } else {
        print('Invalid input');
      }
  }
}
